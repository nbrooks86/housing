CREATE VIEW "CORE_LEVEL1"."HOUSING_ROOMBASE" AS
SELECT
	CAST(ROOMBASEID AS NVARCHAR(8)) AS "ROOMBASE_ID" ,
	CAST(ROOMLOCATIONID AS NVARCHAR(8)) AS "ROOM_LOCATION_ID" ,
	"ROOMLOCATIONFLOORSUITEID" ,
	--"RECORDTYPEENUM" , ONLY NON-ZERO RECORD IS ROOMLCATIONID = 0
	DESCRIPTION AS "ROOMBASE_DESCRIPTION",
	--"ROOMMANAGERID" , ONLY 1 NON-ZERO ENTRY
	--"COMMENTS" , ALL NULL
	--"ALLOCATESORTORDER" ,
	DISABLEDACCESS AS "DISABLED_ACCESS",
	--"AREA" , ALL NULL
	--"BATHROOMS" , ALL BUT 1 NULL
	--"BEDCAPACITY" , ALL BUT 1 ARE 1
	--"FURNITURE" , NULL
	--"FLOORING" , NULL
	--"CUSTOMBIT1" ,NULL
	--"CUSTOMBIT2" ,NULL
	--"CUSTOMSTRING1" ,NULL
	--"CUSTOMSTRING2" ,NULL
	--"CUSTOMSTRING3" ,NULL
	--"CUSTOMSTRING4" ,NULL
	--"CUSTOMSTRING5" ,NULL
	--"CUSTOMSTRING6" ,NULL
	--TO_TIMESTAMP("CUSTOMDATE1") AS "CUSTOMDATE1",NULL
	--TO_TIMESTAMP("CUSTOMDATE2") AS "CUSTOMDATE2",NULL
	TO_TIMESTAMP(BOOKINGHOLDDATETIME) AS "BOOKING_HOLD_DATE_TIME",
	CAST(BOOKINGHOLD_ENTRYID AS NVARCHAR(8)) AS "BOOKING_HOLD_ENTRY_ID",
	TO_TIMESTAMP("DATEMODIFIED") AS "DATE_MODIFIED"
	--"VERSION" ALL 1
FROM "STARREZ"."ROOMBASE"
